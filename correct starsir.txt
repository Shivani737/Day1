using System;
					
public class Program
{
	public static void Main()
	{
		int counter = 5;
	    for(int i = 1; i<=5; i++)
		{
			for(int j = 1; j <=counter-1; j++)
			{
				Console.Write(" ");
			}
			for (int k = 1; k <= 2*i-1; k++)
			{
				Console.Write("*");
			}
			counter--;
			Console.WriteLine();
		}
           
            
	    }
}
